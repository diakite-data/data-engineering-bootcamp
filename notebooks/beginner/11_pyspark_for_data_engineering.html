def transform_data(df):
    """Transforme et nettoie les donnÃ©es"""
    print(" Transformation des donnÃ©es")
    
    # 1. Convertir la date
    df = df.withColumn("date", F.to_date(F.col("date")))
    
    # 2. GÃ©rer les valeurs manquantes
    df = df.fillna({
        "ville": "Inconnu",
        "montant": 0
    })
    
    # 3. Filtrer les donnÃ©es invalides
    df = df.filter(
        (F.col("quantite") > 0) & 
        (F.col("montant") >= 0)
    )
    
    # 4. CrÃ©er des colonnes dÃ©rivÃ©es
    df = df.withColumn(
        "prix_unitaire",
        F.when(F.col("quantite") > 0, F.col("montant") / F.col("quantite")).otherwise(0)
    )
    
    df = df.withColumn(
        "annee",
        F.year(F.col("date"))
    )
    
    df = df.withColumn(
        "mois",
        F.month(F.col("date"))
    )
    
    df = df.withColumn(
        "jour_semaine",
        F.dayofweek(F.col("date"))
    )
    
    # 5. CatÃ©goriser
    df = df.withColumn(
        "categorie_montant",
        F.when(F.col("montant") < 1500, "Faible")
         .when(F.col("montant") < 2000, "Moyen")
         .otherwise("Ã‰levÃ©")
    )
    
    # 6. Ajouter metadata
    df = df.withColumn("processed_at", F.current_timestamp())
    
    print(f" {df.count()} lignes transformÃ©es")
    return df

def aggregate_data(df):

    print(" AgrÃ©gation des donnÃ©es")

    agg_ville_produit = df.groupBy("ville", "produit").agg(

    return {
